35 unsigned int visual_math_round_power_of_2 (unsigned int n)
36 {
37  n--;
38  n |= n >> 1;
39  n |= n >> 2;
40  n |= n >> 4;
41  n |= n >> 8;
42  n |= n >> 16;
43 #if SIZEOF_INT > 4
44  n |= n >> 32;
45 #endif
46  n++;
47 
48  return n;
49 }
