00151   void usleep (uint64_t usecs)
00152   {
00153 #ifdef HAVE_NANOSLEEP
00154       struct timespec request, remaining;
00155       request.tv_sec  = usecs / VISUAL_USEC_PER_SEC;
00156       request.tv_nsec = 1000 * (usecs % VISUAL_USEC_PER_SEC);
00157       while (nanosleep (&request, &remaining) == EINTR)
00158           request = remaining;
00159 #elif defined(HAVE_SELECT)
00160       struct timeval tv;
00161       tv.tv_sec = usecs / VISUAL_USEC_PER_SEC;
00162       tv.tv_usec = usecs % VISUAL_USEC_PER_SEC;
00163       select (0, NULL, NULL, NULL, &tv);
00164 #elif defined(HAVE_USLEEP)
00165       usleep (usecs);
