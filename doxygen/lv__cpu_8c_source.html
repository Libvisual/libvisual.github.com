00092 static void sigill_handler_sse( int signal, struct sigcontext sc )
00093 {
00094     /* Both the "xorps %%xmm0,%%xmm0" and "divps %xmm0,%%xmm1"
00095      * instructions are 3 bytes long.  We must increment the instruction
00096      * pointer manually to avoid repeated execution of the offending
00097      * instruction.
00098      *
00099      * If the SIGILL is caused by a divide-by-zero when unmasked
00100      * exceptions aren't supported, the SIMD FPU status and control
00101      * word will be restored at the end of the test, so we don't need
00102      * to worry about doing it here.  Besides, we may not be able to...
00103      */
00104     sc.eip += 3;
00105 
00106     __lv_cpu_caps.hasSSE=0;
00107 }

00109 static void sigfpe_handler_sse( int signal, struct sigcontext sc )
00110 {
00111     if ( sc.fpstate->magic != 0xffff ) {
00112         /* Our signal context has the extended FPU state, so reset the
00113          * divide-by-zero exception mask and clear the divide-by-zero
00114          * exception bit.
00115          */
00116         sc.fpstate->mxcsr |= 0x00000200;
00117         sc.fpstate->mxcsr &= 0xfffffffb;
00118     } else {
00119         /* If we ever get here, we're completely hosed.
00120         */
00121     }
00122 }
