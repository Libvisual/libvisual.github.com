00098     irnd = (m_impl->seed_state = val_a * m_impl->seed_state + val_c);
00099 
00100 #if VISUAL_RANDOM_FAST_FP_RND
00101     // This saves floating point division (20 clocks on AXP, 38 on P4)

00106     value.i[0] = (irnd << 20);

00109     return value.d - 1.0;

00111     return double (irnd) / VISUAL_RANDOM_MAX;
