00047   {
00048       if (src1.size () != src2.size ())

00053 
00054       for (unsigned int i = 0; i < colors.size (); i++) {
00055           colors[i].r = src1.colors[i].r + ((src2.colors[i].r - src1.colors[i].r) * rate);

00061   VisColor Palette::color_cycle (float rate)
00062   {
00063       int irate = int (rate);
00064 
00065       float rdiff = rate - irate;
00066 
00067       // If rate is exactly an item, return that item
00068       if (rdiff == 0) {
00069           return colors[irate];
00070       }
00071 
00072       irate %= colors.size ();
00073 
00074       uint8_t alpha = rdiff * 255;
00075 
00076       Color const* tmp1 = &colors[irate];
00077       Color const* tmp2;
00078 
00079       if (irate == int (size() - 1))
00080           tmp2 = &colors[0];
00081       else
00082           tmp2 = &colors[irate + 1];
00083 
00084       return Color ((alpha * (tmp1->r - tmp2->r) >> 8) + tmp2->r,
00085                     (alpha * (tmp1->g - tmp2->g) >> 8) + tmp2->g,
00086                     (alpha * (tmp1->b - tmp2->b) >> 8) + tmp2->b);
