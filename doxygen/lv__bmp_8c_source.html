00056 {
00057     uint8_t *data;

00059     int pad;
00060 
00061     pad = (4 - (video->pitch & 3)) & 3;
00062     data = (uint8_t *) visual_video_get_pixels (video) + (video->height * video->pitch);
00063 
00064     switch (depth) {
00065         case 24:
00066         case 8:
00067             while (data > (uint8_t *) visual_video_get_pixels (video)) {
00068                 data -= video->pitch;
00069 
00070                 if (fread (data, video->pitch, 1, fp) != 1)
00071                     goto err;
00072 
00073                 if (pad)
00074                     fseek (fp, pad, SEEK_CUR);
00075             }
00076             break;
00077 
00078         case 4:
00079             while (data > (uint8_t *) visual_video_get_pixels (video)) {
00080                 /* Unpack 4 bpp pixels aka 2 pixels per byte */
00081                 uint8_t *col = data - video->pitch;
00082                 uint8_t *end = (uint8_t *) ((intptr_t)data & ~1);
00083                 data = col;
00084 
00085                 while (col < end) {
00086                     uint8_t p = fgetc (fp);
00087                     *col++ = p >> 4;
00088                     *col++ = p & 0xf;
00089                 }
00090 
00091                 if (video->pitch & 1)
00092                     *col++ = fgetc (fp) >> 4;
00093 
00094                 if (pad)
00095                     fseek (fp, pad, SEEK_CUR);
00096             }
00097             break;
00098 
00099         case 1:
00100             while (data > (uint8_t *) visual_video_get_pixels (video)) {
00101                 /* Unpack 1 bpp pixels aka 8 pixels per byte */
00102                 uint8_t *col = data - video->pitch;

00137 static int load_rle (FILE *fp, VisVideo *video, int mode)
00138 {
00139     uint8_t *col, *end;
00140     uint8_t p;
00141     int c, y, k, pad;
00142     int processing = 1;
00143 
00144     end = (uint8_t *)visual_video_get_pixels (video) + (video->height * video->pitch);
00145     col = end - video->pitch;
00146     y = video->height - 1;
00147 
00148     do {
00149         if ((c = fgetc (fp)) == EOF)
00150             goto err;
00151 
00152         if (c) {
00153             if (y < 0)
00154                 goto err;
00155 
00156             /* Encoded mode */
00157             p = fgetc (fp); /* Color */
00158             if (mode == BI_RLE8) {
00159                 while (c-- && col < end)
00160                     *col++ = p;
00161             } else {
00162                 k = c >> 1; /* Even count */
00163                 while (k-- && col < end - 1) {
00164                     *col++ = p >> 4;
00165                     *col++ = p & 0xf;
00166                 }
00167 
00168                 if (c & 1 && col < end)
00169                     *col++ = p >> 4;
00170             }
00171             continue;
00172         }
00173 
00174         /* Escape sequence */
00175         c = fgetc (fp);
00176         switch (c) {
00177             case EOF:
00178                 goto err;
00179 
00180             case 0: /* End of line */
00181                 y--;
00182                 col = (uint8_t *) visual_video_get_pixels (video) + video->pitch * y;
00183 
00184                 /* Normally we would error here if y < 0.
00185                  * However, some encoders apparently emit an
00186                  * End-Of-Line sequence at the very end of a bitmap.
00187                  */
00188                 break;
00189 
00190             case 1: /* End of bitmap */
00191                 processing = 0;
00192                 break;
00193 
00194             case 2: /* Delta */
00195                 /* X Delta */
00196                 col += (uint8_t) fgetc (fp);
00197 
00198                 /* Y Delta */
00199                 c = (uint8_t) fgetc (fp);
00200                 col -= c * video->pitch;
00201                 y -= c;
00202 
00203                 if (col < (uint8_t *)visual_video_get_pixels (video))
00204                     goto err;
00205 
00206                 break;
00207 
00208             default: /* Absolute mode: 3 - 255 */
00209                 if (mode == BI_RLE8) {
00210                     pad = c & 1;
00211                     while (c-- && col < end)
00212                         *col++ = fgetc (fp);
00213                 } else {
00214                     pad = ((c + 1) >> 1) & 1;
00215                     k = c >> 1; /* Even count */
00216                     while (k-- && col < end - 1) {
00217                         p = fgetc (fp);
00218                         *col++ = p >> 4;
00219                         *col++ = p & 0xf;
00220                     }
00221 
00222                     if (c & 1 && col < end)
00223                         *col++ = fgetc (fp) >> 4;
00224                 }
00225 
00226                 if (pad)
00227                     fgetc (fp);
00228                 break;
00229 
00230         }
00231     } while (processing);
00232 
00233     return VISUAL_OK;
00234 
00235 err:
00236     visual_log (VISUAL_LOG_ERROR, _("Bitmap data is not complete"));
00237 
00238     return -VISUAL_ERROR_BMP_CORRUPTED;
00239 }
