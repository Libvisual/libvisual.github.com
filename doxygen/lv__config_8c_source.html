00088     VisConfigRegistry *registry;
00089     VisConfigRegistrySection *rsection;
00090     int fd;
00091     int length;
00092     uint32_t datalength;
00093     char namebuf[64];

00101 
00102     /* Opening file */
00103     fd = open (configfile, O_RDONLY);
00104 
00105     if (fd < 0)
00106         goto out;
00107 
00108     length = lseek (fd, 0, SEEK_END);
00109 
00110     lseek (fd, 0, SEEK_SET);
00111 
00112     /* Empty config registry */
00113     if (length == 0)
00114         goto out;
00115 
00116     /* Checking version */
00117     if (read (fd, namebuf, 19) != 19)
00118         goto broken;
00119 
00120     /* Different config registry version, won't load */
00121     if (strncmp (namebuf, VISUAL_CONFIG_VERSION, 19) != 0) {
00122         visual_log (VISUAL_LOG_WARNING, _("The config registry file format is of an obsolete file format, won't load it"));
00123 
00124         goto out;
00125     }
00126 
00127     /* Loading sections */
00128     while (lseek (fd, 1, SEEK_CUR) > 0) {
00129         lseek (fd, -1, SEEK_CUR);
00130 
00131         if (read (fd, &datalength, sizeof (uint32_t)) != 4)
00132             goto broken;
00133 
00134         if (read (fd, sectionname, datalength > 128 ? 128 : datalength) != (datalength > 128 ? 128 : datalength))
00135             goto broken;
00136 
00137         sectionname[127] = '\0';
00138 
00139         rsection = visual_config_registry_section_new ();
00140 
00141         rsection->name = visual_strdup (sectionname);
00142 
00143         lseek (fd, (datalength > 128 ? -128 : -datalength) + strlen (sectionname), SEEK_CUR);
00144 
00145         rsection->datalength = datalength;
00146         rsection->data = visual_mem_malloc0 (datalength);
00147 
00148         if (read (fd, rsection->data, datalength) != datalength) {
00149             visual_object_unref (VISUAL_OBJECT (rsection));
00150 
00151             goto broken;
00152         }
00153 
00154         visual_config_registry_add (registry, rsection);
00155     }
00156 
00157 broken:
00158     visual_log (VISUAL_LOG_ERROR, _("Broken config registry, won't load"));
00159 
00160     /* Unload all sections, some might have been partially parsed, which can be dangerious, we don't
00161      * want that. */
00162     visual_collection_destroy (VISUAL_COLLECTION (&registry->sections));
00163 
00164 out:
