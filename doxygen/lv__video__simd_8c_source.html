00028 int _lv_blit_overlay_alphasrc_mmx (VisVideo *dest, VisVideo *src)
00029 {
00030 #if defined(VISUAL_ARCH_X86) || defined(VISUAL_ARCH_X86_64)
00031     int i, j;
00032     uint8_t *destbuf = visual_video_get_pixels (dest);
00033     uint8_t *srcbuf = visual_video_get_pixels (src);
00034 
00035     for (i = 0; i < src->height; i++) {
00036         for (j = 0; j < src->width; j++) {
00037             __asm __volatile
00038                 ("\n\t movd %[spix], %%mm0"
00039                  "\n\t movd %[dpix], %%mm1"
00040                  "\n\t movq %%mm0, %%mm2"
00041                  "\n\t movq %%mm0, %%mm3"
00042                  "\n\t psrlq $24, %%mm2"    /* The alpha */
00043                  "\n\t movq %%mm0, %%mm4"
00044                  "\n\t psrld $24, %%mm3"
00045                  "\n\t psrld $24, %%mm4"
00046                  "\n\t psllq $32, %%mm2"
00047                  "\n\t psllq $16, %%mm3"
00048                  "\n\t por %%mm4, %%mm2"
00049                  "\n\t punpcklbw %%mm6, %%mm0"  /* interleaving dest */
00050                  "\n\t por %%mm3, %%mm2"
00051                  "\n\t punpcklbw %%mm6, %%mm1"  /* interleaving source */
00052                  "\n\t psubsw %%mm1, %%mm0" /* (src - dest) part */
00053                  "\n\t pmullw %%mm2, %%mm0" /* alpha * (src - dest) */
00054                  "\n\t psrlw $8, %%mm0"     /* / 256 */
00055                  "\n\t paddb %%mm1, %%mm0"  /* + dest */
00056                  "\n\t packuswb %%mm0, %%mm0"
00057                  "\n\t movd %%mm0, %[dest]"
00058                  : [dest] "=m" (*destbuf)
00059                  : [dpix] "m" (*destbuf)
00060                  , [spix] "m" (*srcbuf));
00061 
00062             destbuf += 4;
00063             srcbuf += 4;
00064         }
00065 
00066         destbuf += dest->pitch - (dest->width * dest->bpp);
00067         srcbuf += src->pitch - (src->width * src->bpp);
00068     }
00069 
00070     return VISUAL_OK;
00071 #else /* !VISUAL_ARCH_X86 */
00072     return VISUAL_ERROR_CPU_INVALID_CODE;

00076 int _lv_scale_bilinear_32_mmx (VisVideo *dest, VisVideo *src)
00077 {
00078 #if defined(VISUAL_ARCH_X86) || defined(VISUAL_ARCH_X86_64)
00079     uint32_t y;
00080     uint32_t u, v, du, dv; /* fixed point 16.16 */
00081     uint32_t *dest_pixel, *src_pixel_rowu, *src_pixel_rowl;
00082 
00083     dest_pixel = visual_video_get_pixels (dest);
00084 
00085     du = ((src->width - 1)  << 16) / dest->width;
00086     dv = ((src->height - 1) << 16) / dest->height;
00087     v = 0;
00088 
00089     for (y = dest->height; y--; v += dv) {
00090         uint32_t x;
00091         uint32_t fracU, fracV;     /* fixed point 28.4 [0,1[    */
00092 
00093         if (v >> 16 >= src->height - 1)
00094             v -= 0x10000;
00095 
00096         src_pixel_rowu = (src->pixel_rows[v >> 16]);
00097         src_pixel_rowl = (src->pixel_rows[(v >> 16) + 1]);
00098 
00099         /* fracV = frac(v) = v & 0xffff */
00100         /* fixed point format convertion: fracV >>= 8) */
00101         fracV = ((v & 0xffff) >> 12) | 0x100000;
00102         u = 0;
00103 
00104         for (x = dest->width - 1; x--; u += du) {
00105 
00106             /* fracU = frac(u) = u & 0xffff */
00107             /* fixed point format convertion: fracU >>= 8) */
00108             fracU  = ((u & 0xffff) >> 12) | 0x100000;
00109 
00110             __asm__ __volatile__
00111                 ("\n\t pxor %%mm7, %%mm7"
00112                  /* Prefetching does not show improvement on my Duron (maybe due to its small cache?) */
00113                  /*"\n\t prefetch  64%[pixel_l]" / * only work on 3now!/SSE cpu */
00114                  /*"\n\t prefetchw 64%[output]"  / * only work on 3now!/SSE cpu */
00115 
00116                  /* Computing coefs values (Thread #1 and #2) => ends on #C
00117                   *
00118                   * notice 0x10 = 1.0 (fixed point 28.4 - like fracU and fracV)
00119                   *
00120                   * coef[0] = (0x10 - fracU) * (0x10 - fracV); * UL=0 *
00121                   * coef[1] = (0x10 - fracU) * fracV;          * LL=1 *
00122                   * coef[2] = fracU * (0x10 - fracV);          * UR=2 *
00123                   * coef[3] = fracU * fracV;                   * LR=3 *
00124                   */
00125 
00126                  /*
00127                   * Unpacking colors (Thread #3 and #4)
00128                   */
00129                  /*
00130                   * Multiplying colors by coefs (Threads #5 and #6)
00131                   */
00132                  /*
00133                   * Adding colors together. (Thread #7)
00134                   */
00135 
00136                 "#1\n\t movd %[fracu], %%mm4"   /* mm4 = [ 0 | 0 | 0x10 | fracU ] */
00137                 "#2\n\t movd %[fracv], %%mm6"   /* mm6 = [ 0 | 0 | 0x10 | fracV ] */
00138 
00139                 "#1\n\t punpcklwd %%mm4, %%mm4" /* mm4 = [ 0x10 | 0x10 | fracU | fracU ] */
00140                 "#2\n\t movq      %%mm6, %%mm3"
00141 
00142                 "#1\n\t pxor      %%mm5, %%mm5"
00143                 "#2\n\t punpckldq %%mm6, %%mm6" /* mm6 = [ 0x10 | fracv | 0x10 | fracV ] */ 
00144                 "#3\n\t movq %[pixel_u], %%mm0" /* mm0 = [ col[0] | col[2] ] */
00145 
00146                 "#1\n\t punpckldq %%mm4, %%mm5" /* mm5 = [ fracU | fracU | 0 | 0 ] */
00147                 "#2\n\t punpcklwd %%mm7, %%mm3" /* mm3 = [ 0    | 0x10  | 0    | fracV ] */
00148                 "#3\n\t movq      %%mm0, %%mm2"
00149 
00150                 "#1\n\t psubusw   %%mm5, %%mm4" /* mm4 = [ 0x10-fracU | 0x10-fracU | fracU | fracU ] */
00151                 "#2\n\t punpckldq %%mm3, %%mm3" /* mm3 = [ 0    | fracV | 0    | fracV ] */

00153 
00154                 "#2\n\t pslld     $16,   %%mm3" /* mm3 = [ fracV | 0 | fracV | 0 ] */
00155                 "#3\n\t punpcklbw %%mm7, %%mm0" /* mm0 = [ col[0] unpacked ] */
00156 
00157                 "#2\n\t psubusw   %%mm3, %%mm6" /* mm6 = [ 0x10-fracV | fracV | 0x10-fracV | fracV ] */
00158                 "#4\n\t movq      %%mm1, %%mm3"
00159 
00160                 "#C\n\t pmullw    %%mm6, %%mm4" /* mm4 = [ coef[0]|coef[1]|coef[2]|coef[3] ] */
00161                 "#5\n\t movq      %%mm4, %%mm5"

00201             ++dest_pixel;
00202         }
00203 
00204         dest_pixel += (dest->pitch / 4) - ((dest->width - 1));
00205     }

00207     __asm__ __volatile__ ("\n\t emms");
