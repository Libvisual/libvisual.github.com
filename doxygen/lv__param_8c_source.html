00072 static int get_next_pcall_id (VisList *callbacks)
00073 {
00074     VisListEntry *le = NULL;
00075     VisParamEntryCallback *pcall;
00076     int found = FALSE;
00077     int i;
00078 
00079     /* Walk through all possible ids */
00080     for (i = 0; i < VISUAL_PARAM_CALLBACK_ID_MAX; i++) {
00081 
00082         found = FALSE;
00083         /* Check all the callbacks if the id is used */
00084         while ((pcall = visual_list_next (callbacks, &le)) != NULL) {
00085 
00086             /* Found the ID, break and get ready for the next iterate */
00087             if (pcall->id == i) {
00088                 found = TRUE;
00089 
00090                 break;
00091             }
00092         }
00093 
00094         /* The id has NOT been found, thus is an original, and we return this as the next id */
00095         if (!found)
00096             return i;
00097     }
00098 
00099     /* This is virtually impossible, or something very wrong is going ok, but no id seems to be left */
00100     return -1;
00101 }
